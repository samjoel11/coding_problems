import java.util.*;
import java.io.*;

class Loops2{
    public static void main(String []argh){
        Scanner in = new Scanner(System.in);
        // t denotes the number of queries
        int t = in.nextInt();
        for(int i = 0; i < t; i++){
            int a = in.nextInt();
            int b = in.nextInt();
            int n = in.nextInt();
            
            int currentSum = a;
            for(int j = 0; j < n; j++){
                // Use Math.pow to calculate 2^j and multiply by b
                currentSum += (int)Math.pow(2, j) * b;
                System.out.print(currentSum + " ");
            }
            // Move to a new line after each query's series is printed
            System.out.println();
        }
        in.close();
    }
}
